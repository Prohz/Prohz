        public List<ResponseModel> GetData(string bussinessname,string bussinessaddress)
        {
           
            List<ResponseModel> responsemodel = new List<ResponseModel>();
          
            HtmlAgilityPack.HtmlWeb web = new HtmlAgilityPack.HtmlWeb();
            HtmlAgilityPack.HtmlDocument doc = web.Load("https://www.yellowpages.com/search?search_terms=" + bussinessname + "&geo_location_terms="+ bussinessaddress);
            HtmlNode[] nodes = doc.DocumentNode.SelectNodes("//div[@class='v-card']").ToArray();
              foreach (HtmlNode item in nodes)
                {
                //var namez = item.SelectSingleNode(".//img[@src]")?.Attributes["src"].Value; 
                ResponseModel response = new ResponseModel() { 

                    name = item.SelectSingleNode(".//a[@class='business-name']")?.InnerText,
                    categories = item.SelectSingleNode(".//div[@class='categories']")?.InnerText,
                    FullAddress = item.SelectSingleNode(".//p[@class='adr']")?.InnerText,
                    Street= item.SelectSingleNode(".//div[@class='street-address']")?.InnerText,
                    locality = item.SelectSingleNode(".//div[@class='street-locality']")?.InnerText,
                    imgurl = item.SelectSingleNode(".//img[@src]")?.Attributes["src"].Value,
                    phone = item.SelectSingleNode(".//div[@class='phones']")?.InnerText,
                    open= item.SelectSingleNode(".//div[@class='open-status']")?.InnerText,


                };
                responsemodel.Add(response);


                }

           


            return responsemodel;

        }
